#事件命名规范
#蠢驴的event id分为两部分，<namespace>.<integer ID>
#前一部分是字母和下划线，后一部分是整数
#通常情况下，我们习惯通过整数位数表示子事件
#例如，touhou.11表示touhou.1的子事件
#但你很容易发现，touhou.11还可以表示第11个一级事件
#为保证代码可维护性，需要积极使用add_namespace增加<namespace>
#具体而言，一个<namespace>每一级只允许有1-9共9个同级事件
#关于id为什么不使用“0”
#我讲一下p社的事件机制
#<namespace>是一个“域”，<integer ID>是具体值
#对，你没有看错，事件id不是独立识别的字符串，而是数值
#这就导致0、00本质上是相同的，所以我禁止使用“0”
#add_namespace使用规范
#add_namespace = a_b_c_d，a、b、c、d代表各级标题
#事件位置顺序
#本地化的位置顺序应该与事件本身的位置顺序相同，事件本身的位置顺序应该按事件链逻辑，而不是命名顺序
#例如，1=小红吃饭，2=小红做饭，2应该在1前面，这样有助于减少后期维护逻辑思维量
#子事件直接在上级事件后面
#例如，21=小红把饭端上桌子，整体顺序：2→21→1
#幻想乡统一事件

	title = touhou_devildom.5.t

	is_triggered_only = yes

		name = touhou_devildom.5.a
		ai_chance = {
			base = 0
		}
		add_timed_idea = {
			idea = touhou_god_of_makai_ideas_preparation
			days = 360
		}
		custom_effect_tooltip = touhou_devildom_coming_one_country
		hidden_effect = {
			if = {
				#优先与玩家关系不密切的ai南非国家
				limit = {
					any_other_country = {
						owns_state = 681
						is_ai = yes
						exists = yes
						NOT = {
							is_in_faction_with = ROOT
						}						#优先没有与玩家结盟的国家
						NOT = {
							is_subject_of = ROOT
						}						#优先不是玩家傀儡的国家
					}
				}
				random_other_country = {
					limit = {
						owns_state = 681
						is_ai = yes
						exists = yes
						NOT = {
							is_in_faction_with = ROOT
						}						#优先没有与玩家结盟的国家
						NOT = {
							is_subject_of = ROOT
						}						#优先不是玩家傀儡的国家
					}
					country_event = {
						id = touhou_devildom.1
					}
				}
			}
			else = {
				random_other_country = {
					limit = {
						is_ai = yes
						exists = yes
						NOT = {
							is_in_faction_with = ROOT
						}

							is_subject_of = ROOT
						}
					}
					country_event = {
						id = touhou_devildom.1
					}
				}

		}
	}

		name = touhou_devildom.5.b

			base = 1
		}
	}
}
